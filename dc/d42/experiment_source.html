106 ## 指定したクラスを動的にimportする関数
107 #
108 # @param pyfile_str 対象とするpythonスクリプトのパス文字列
109 # @param class_str template.Mainを継承したクラス名の文字列
110 # @param logger ロガーオブジェクト
111 # @return クラス名とクラスを保持するリスト
112 def dynamic_import(pyfile_str, class_str, logger=getLogger()):
113  abspath = os.path.abspath(pyfile_str)
114  abspath_without_ext = os.path.splitext(abspath)[0]
115  dir, module_name = os.path.split(abspath_without_ext)
116  class_name = class_str
117  logger.info('direcotry of target script: %s' % dir)
118  logger.info('module name of target script: %s' % module_name)
119  logger.info('name of target class: %s' % class_name)
120  sys.path.append(dir)
121  logger.info('modulename: %s' % module_name)
122  try:
123  class_ = getattr(__import__(module_name), class_name)
124  logger.info('target class has been loaded')
125  except ImportError as e:
126  logger.exception(e)
127  logger.error('No module named %s' % module_name)
128  class_name = '%s.%s' % (module_name, class_name)
129  return [class_name, class_]
